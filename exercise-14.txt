# Reverse all fax phone numbers, keeping the phone numbers in a valid format
# List all of the new fax phone number values


mysql>  SELECT 
            CONCAT_WS('-', '1', REVERSE(lineNumS), REVERSE(lineNumL), REVERSE(areaCode)) AS faxNum,
            subtype,
            content
        FROM (SELECT
            SUBSTRING(REPLACE(RIGHT(content, 8), '-', ''), 1, 4) AS lineNumL,
            SUBSTRING(REPLACE(RIGHT(content, 8), '-', ''), 5, 3) AS lineNumS,
            CASE SUBSTRING(content, 1, 1)
                WHEN '(' THEN SUBSTRING(content, 2, 3)
                WHEN '1' THEN SUBSTRING(content, 3, 3)
                
            END
            AS areaCode,
            subtype,
            content
        FROM
            Phone
        WHERE
            subtype="fax") AS tablesplit;
