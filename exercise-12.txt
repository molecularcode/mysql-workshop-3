# List 100 phone numbers in separated parts: country code, area code and line number


mysql>  SELECT
            "1" AS countyCode,
            CASE SUBSTRING(content, 1, 1)
                WHEN '(' THEN SUBSTRING(content, 2, 3)
                WHEN '1' THEN SUBSTRING(content, 3, 3)
                
            END
            AS areaCode,
            RIGHT(content, 8) AS lineNum
        FROM
            Phone
        LIMIT 100;
